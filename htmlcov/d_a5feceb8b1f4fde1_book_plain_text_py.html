29 def find_nearest_page_end_match( 

30 self, page_start_index: int, pattern: re.Pattern[str] 

31 ) -> Optional[int]: 

32 """Find the nearest regex match around expected end of page. 

33 

34 In no such match in the vicinity, return None. 

35 Calculate the vicinity based on the expected PAGE_LENGTH_TARGET 

36 and PAGE_LENGTH_ERROR_TOLERANCE. 

37 """ 

38 end_pos = min( 

39 page_start_index 

40 + int(self.PAGE_LENGTH_TARGET * (1 + self.PAGE_LENGTH_ERROR_TOLERANCE)), 

41 self.book_end, 

42 ) 

43 start_pos = max( 

44 page_start_index 

45 + int(self.PAGE_LENGTH_TARGET * (1 - self.PAGE_LENGTH_ERROR_TOLERANCE)), 

46 0, 

47 ) 

48 ends = [match.end() for match in pattern.finditer(self.text, start_pos, end_pos)] 

49 return ( 

50 min(ends, key=lambda x: abs(x - page_start_index + self.PAGE_LENGTH_TARGET)) 

51 if ends 

52 else None 

53 )

55 def find_nearest_page_end(self, page_start_index: int) -> int: 

56 """Find the nearest page end.""" 

57 patterns = [ # sorted by priority 

58 re.compile(r"\r?\n(\s*\r?\n)+"), # Paragraph end 

59 re.compile(r"[^\s.!?]*\w[^\s.!?]*[.!?]\s"), # Sentence end 

60 re.compile(r"\w+\b"), # Word end 

61 ] 

62 

63 for pattern in patterns: 

64 if nearest_page_end := self.find_nearest_page_end_match(page_start_index, pattern): 

65 # nearest_page_end = self.adjust_heading(page_start_index, nearest_page_end) 

66 return nearest_page_end 

67 

68 # If no suitable end found, return the maximum allowed length 

69 return min(self.book_end, page_start_index + self.PAGE_LENGTH_TARGET)

72 def normalize(text: str) -> str: 

73 """Make later processing more simple.""" 

74 text = re.sub(r"\r?\n(\s*\r?\n)+", "\n\n", text) 

75 text = re.sub(r"\r", "", text) 

76 text = re.sub(r"[^\S\n\r]+", " ", text) 

77 return text
