126 @classmethod 

127 def update_user_progress( 

128 cls, user: CustomUser, book_id: int, page_number: int, top_word_id: int 

129 ) -> None: 

130 """Update user reading progress.""" 

131 reader_profile, _ = ReaderProfile.objects.get_or_create(user=user) 

132 

133 # We search using user&book only but this is not enough to create a new record. 

134 # Thus we cannot use get_or_create. 

135 reading_progress = cls.objects.filter( 

136 reader=reader_profile, book_id=book_id 

137 ).first() or cls(reader=reader_profile, book_id=book_id) 

138 reading_progress.page_number = page_number 

139 reading_progress.top_word_id = top_word_id 

140 reading_progress.last_read_time = timezone.now() 

141 reading_progress.save()
