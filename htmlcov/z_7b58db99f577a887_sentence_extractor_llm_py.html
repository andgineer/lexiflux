26def break_into_sentences_llm( # pylint: disable=too-many-locals 

27 plain_text: str, 

28 word_slices: List[Tuple[int, int]], 

29 term_word_ids: List[int], # pylint: disable=redefined-outer-name 

30 lang_code: str = "en", 

31) -> Tuple[List[str], Dict[int, int]]: 

32 """ 

33 Break plain text into sentences, mark the highlighted word, and map word IDs 

34 to sentence indices. 

35 

36 Args: 

37 plain_text: The input text without HTML tags. 

38 word_slices: List of word start and end indices. 

39 term_word_ids: The ID of the word to be highlighted. 

40 lang_code: Language code of the text. 

41 

42 Returns: 

43 Tuple[List[str], Dict[int, int]]: A tuple containing: 

44 - List of sentences 

45 - Dictionary mapping word index to sentence index 

46 """ 

47 # Step 1: Mark the highlighted term with ** 

48 term_start = word_slices[term_word_ids[0]][0] 

49 term_end = word_slices[term_word_ids[-1]][1] 

50 marked_text = ( 

51 plain_text[:term_start] 

52 + WORD_START_MARK 

53 + plain_text[term_start:term_end] 

54 + WORD_END_MARK 

55 + plain_text[term_end:] 

56 ) 

57 

58 # Step 2: Use LLM to detect and mark the sentence 

59 marked_text_with_sentence = detect_sentence_llm(marked_text, lang_code) 

60 

61 # Step 3: Extract sentences and create word_to_sentence mapping 

62 sentences_list = [] 

63 map_word_to_sentence = {} 

64 current_sentence = 0 

65 

66 sentence_start = marked_text_with_sentence.find(SENTENCE_START_MARK) 

67 if sentence_start == -1: 

68 return [], {} 

69 sentence_end = marked_text_with_sentence.find( 

70 SENTENCE_END_MARK, sentence_start + len(SENTENCE_START_MARK) 

71 ) 

72 if sentence_end == -1: 

73 return [], {} 

74 

75 # Extract the sentence and remove WORD_START_MARK and WORD_END_MARK 

76 marked_sentence = marked_text_with_sentence[ 

77 sentence_start + len(SENTENCE_START_MARK) : sentence_end 

78 ] 

79 clean_sentence = marked_sentence.replace(WORD_START_MARK, "").replace(WORD_END_MARK, "") 

80 sentences_list.append(clean_sentence) 

81 # sentence end index after removing word and sentence marks 

82 sentence_end -= len(SENTENCE_START_MARK) + len(WORD_START_MARK) + len(WORD_END_MARK) 

83 

84 for word_id, (word_start, word_end) in enumerate(word_slices): # pylint: disable=redefined-outer-name 

85 if sentence_start <= word_start < sentence_end: 

86 # print(f"Word ID {word_id} (started at {word_start}) is in sentence: 

87 # {plain_text[0:word_start]}@@@{plain_text[word_start:word_end]} 

88 # @@@{plain_text[word_end:]}") 

89 map_word_to_sentence[word_id] = current_sentence 

90 word_slices[word_id] = (word_start, word_end) 

91 

92 return sentences_list, map_word_to_sentence

95def detect_sentence_llm(text: str, text_language: str) -> str: 

96 """Use LLM to detect and mark the sentence containing the highlighted word.""" 

97 preprocessing_template = ( 

98 "Given a text in {text_language}, identify the full sentence" 

99 "that contains the word(s) marked with double asterisks **like this**." 

100 "Mark the sentence containing the marked word(s) with double vertical bars before" 

101 "and after the sentence, like this:" 

102 f"{SENTENCE_START_MARK}This is the marked sentence " 

103 f"containing {WORD_START_MARK}the word{WORD_END_MARK}.{SENTENCE_END_MARK}" 

104 "Definition of a sentence: grammatical unit that consists of one or more words," 

105 "usually containing a subject and a predicate, and expresses a complete thought." 

106 "It typically begins with a capital letter and ends with a terminal punctuation mark." 

107 "Can be simple, containing a single independent clause " 

108 "(e.g., 'The cat sleeps.'), or complex," 

109 "containing one or more dependent clauses in addition to an independent clause" 

110 "(e.g., 'Although it was raining, we went for a walk.')." 

111 "In identifying the sentence, ensure to include any dependent or subordinate clauses" 

112 "that are part of the main thought." 

113 "Return the result as plain text with the marked sentence." 

114 ) 

115 

116 preprocessing_prompt_template = ChatPromptTemplate.from_messages( 

117 [("system", preprocessing_template), ("user", "{text}")] 

118 ) 

119 

120 model = ChatOpenAI( 

121 model="gpt-3.5-turbo", 

122 temperature=0.5, 

123 ) 

124 

125 chain = preprocessing_prompt_template | model | TextOutputParser() 

126 result = chain.invoke({"text": text, "text_language": text_language}) 

127 

128 return result
